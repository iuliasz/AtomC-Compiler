// the if condition must be a scalar value
if("abc");


// the while condition must be a scalar value
while("A");

// a void function cannot return a value
void k(){
    return 1;
}

// the return value must be a scalar value
int k(){
    return "ab";
}

// cannot convert the return expression type to the function return type
struct S s;

int g() {
    return s;
}

// a non-void function must return a value
int g(){
    return;
}

__________________________

// the assign destination must be a left-value
1=x;

// the assign destination cannot be constant
text=1;

// the assign source must be a scalar
int x;
int v[3];

x=v;

// the assign destination must be scalar
int i[10];

void q() {
    i = 3;
}

// the assign source cannot be converted to destination
text[i]=a;

___________________________


// invalid operand type for ||
if(a || v);

___________________________

// invalid operand type for &&
if(a && v);

___________________________

// invalid operand type for == or !=
void g(){
     a==1;
}

__________________________

// invalid operand type for <, <=, >, >=
a<1;

__________________________

// invalid operand type for + or -
a+1;

__________________________

// invalid operand type for * or /
a*2

___________________________

// cannot convert to a struct type
int g(){
	return (struct S) 1;
}

// cannot convert a struct
(int) a;

// an array can be converted only to another array
(char)text;

// a scalar can be converted only to another scalar
(int[])x;

____________________________

// unary - or ! must have a scalar operand
!text;

____________________________

// only an array can be indexed
x[0];

// the index is not convertible to int
text["ab"];

// a field can only be selected from a struct
x.n;

// the structure %s does not have a field %s
a.x;

____________________________

// undefined id: %s
i=x;

// only a function can be called
a();

// too many arguments in function call
void main(){
	h(1,2,3);
}

// in call, cannot convert the argument type to the parameter type
void main(){
	h("ab","ab");
}

// too few arguments in function call
void main(){
	h(1);
}

// a function can only be called
void main(){
	h;
}
